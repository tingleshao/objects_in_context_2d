   def refresh()
     paint
     @app.click do |button, left, top|
       if @inCircle == false
         @atoms << [left, top, 10]
         @inCircle = true
         paint
       else 
         @atoms.last[2] = @atoms.last[2] + 1.5
         paint
       end
   end
   @app.keypress do |k|
     if "#{k.inspect}".to_s == "g"
     else
       @inCircle = false
     end
   end
 end
